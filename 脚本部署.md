## 前提准备

安装离线下载命令

```shell
yum install wget
```

赋予脚本权限，否则有些命令无法执行

```shell
chmod 755 install.sh
```

如果想外部访问，需要开启端口，脚本默认没有开启，需要你手动开启，注意虚拟机和云服务的端口开启方式不一致

```shell
// 开启8080端口，默认端口有
firewall-cmd --zone=public --add-port=8080/tcp --permanent

//  重启端口生效
systemctl restart firewalld
```

## 创建脚本

创建脚本： ` install.sh` 

```sh
#!/bin/bash

function install_java() {
  if command -v java >/dev/null 2>&1; then
    echo "================= 检测到 java 已经安装========================"
  else
    echo "##########	开始安装 Java8	##########"
    yum -y install java-1.8.0-openjdk.x86_64
    echo "##########	Java8 安装完成	##########"
  fi
}

function install_tomcat() {
  if command -v tomcat >/dev/null 2>&1; then
    echo "================= 检测到 Tomcat 已经安装========================"
  else
    echo "##########	开始下载Tomcat 	##########"
    wget http://archive.apache.org/dist/tomcat/tomcat-10/v10.0.22/bin/apache-tomcat-10.0.22.tar.gz
    echo "================= 开始解压========================"
    tar -zxvf apache-tomcat-10.0.22.tar.gz -C /usr/local
    cd /usr/local/apache-tomcat-10.0.22
    echo "================= 开始启动Tomcat========================"
    ./bin/startup.sh
  fi
}

function install_maven() {
  if command -v mvn >/dev/null 2>&1; then
    echo "================= 检测到 Maven 已经安装，跳过安装过程========================"
  else
    echo "================= 下载Maven ========================"
    wget https://repo.huaweicloud.com/apache/maven/maven-3/3.6.3/binaries/apache-maven-3.6.3-bin.tar.gz
    echo "================= 开始解压========================"
    tar -zxvf apache-maven-3.6.3-bin.tar.gz -C /usr/local
    echo "================= 配置 Maven 环境变量========================"
    echo "export PATH=/usr/local/apache-maven-3.6.3/bin:\$PATH" >> /etc/profile
    source /etc/profile
    echo "================= Maven 安装完成========================"
    mvn --version >/dev/null 2>&1
  fi
}

function install_node() {
  if command -v node >/dev/null 2>&1; then
    echo "================= 检测到 Node 已经安装，跳过安装过程========================"
  else
    echo "================= 下载Node ========================"
    wget https://nodejs.org/dist/v12.20.0/node-v12.20.0-linux-x64.tar.gz
    echo "================= 开始解压========================"
    tar -zxvf node-v12.20.0-linux-x64.tar.gz -C /usr/local
    echo "================= 配置 Node 环境变量========================"
    echo "export PATH=/usr/local/node-v12.20.0-linux-x64/bin:\$PATH" >> /etc/profile

    source /etc/profile
    echo "================= Node 安装完成========================"
    node -v >/dev/null 2>&1
    echo "================= 淘宝加速NPM========================"
    npm install cnpm -g --registry=https://registry.npm.taobao.org
  fi
}

function install_nginx() {
  if command -v node >/dev/null 2>&1; then
    echo "================= 检测到 Nginx 已经安装，跳过安装过程========================"
  else
    echo "================= 安装前提插件 ========================"
    yum install gcc-c++ pcre pcre-devel zlib zlib-devel openssl openssl-devel

    wget http://nginx.org/download/nginx-1.18.0.tar.gz

    echo "================= 开始解压========================"
    tar -zxvf nginx-1.18.0.tar.gz -C /usr/local
    cd /usr/local/nginx-1.18.0

    echo "================= 开始安装模块 ========================"
    ./configure \
      --user=nginx \
      --group=nginx \
      --prefix=/usr/local/nginx-1.18.0 \
      --with-http_stub_status_module \
      --with-http_ssl_module \
      --with-http_realip_module \
      --with-http_gzip_static_module
    echo "================= 编译安装 ========================"
    make
    make && install
    echo "================= 设置Nginx 环境变量 ========================"
    echo "export PATH=$PATH:/usr/local/nginx/sbin" >>/etc/profile
    export PATH=$PATH:'/usr/local/nginx/sbin'
    echo "=========== 启动 ==============================="
    /usr/local/nginx/sbin/nginx
    echo "=========== 开机启动 ==============================="
    chmod +x /etc/rc.d/rc.local
  fi

}

while [[ '1'='1' ]]; do
  echo "=================================================="
  echo " \            Linux install.sh 环境部署脚本 V 1.1     / "
  echo "=================================================="
  echo "1：安装Jdk1.8"
  echo "2：安装Node"
  echo "3：安装Tomcat"
  echo "4：安装Maven"
  echo "5: 安装Nginx"
  echo "6: 安装MySQL"
  echo "7：安装Redis"

  echo "0安装全部"
  echo "q.退出"
  read -p "请选择操作：" options
  if [ "$options" == 1 ]; then
    install_java
  elif [ "$options" == 2 ]; then
    install_node
  elif [ "$options" == 3 ]; then
    install_tomcat
  elif [ "$options" == 4 ]; then
    install_maven
  elif [ "$options" == 5 ]; then
    install_nginx
  elif [ "$options" == 6 ]; then
    install_mysql
  elif [ "$options" == 7 ]; then
    install_redis
  elif [ "$options" == 0 ]; then
    echo "##########	安装全部	##########"
    install_java
    install_nginx
    echo "##########	安装完成	##########"

  elif [ "$options" == 'q' ]; then
    exit
  else
    echo "参数错误"
    exit
  fi
done
```
